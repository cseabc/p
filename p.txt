//POSIX threaDDD
#include <stdio.h>
#include <pthread.h>

void* func(void* arg) {
    printf("Hello from thread %d!\n", *(int*)arg);
    return NULL;
}

int main() {
    pthread_t threads[3];
    int ids[3] = {0, 1, 2};
    for (int i = 0; i < 3; i++) pthread_create(&threads[i], NULL, func, &ids[i]);
    for (int i = 0; i < 3; i++) pthread_join(threads[i], NULL);
    return 0;
}
